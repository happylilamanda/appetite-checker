        // Function to parse CSV data into an array of objects

        function parseCSV(csv) {

            const lines = csv.split('\n');

            const headers = lines[0].split(',').map(header => header.trim()); // Trim headers


            const data = [];

            for (let i = 1; i < lines.length; i++) {

                const currentLine = lines[i].trim();

                if (currentLine === '') continue; // Skip empty lines


                // Handle quoted fields

                const row = [];

                let inQuote = false;

                let field = '';

                for (let j = 0; j < currentLine.length; j++) {

                    const char = currentLine[j];

                    if (char === '"' && (j === 0 || currentLine[j-1] === ',' || inQuote)) {

                        inQuote = !inQuote;

                        // Handle escaped quotes (e.g., "" within a field)

                        if (inQuote && j < currentLine.length - 1 && currentLine[j+1] === '"') {

                            field += '"';

                            j++; // Skip the next quote as it's an escaped one

                        }

                    } else if (char === ',' && !inQuote) {

                        row.push(field.trim()); // Trim content of field

                        field = '';

                    } else {

                        field += char;

                    }

                }

                row.push(field.trim()); // Push the last field


                const obj = {};

                headers.forEach((header, index) => {

                    obj[header] = row[index] || ''; // Assign value, default to empty string if missing

                });

                data.push(obj);

            }

            return data;

        }
